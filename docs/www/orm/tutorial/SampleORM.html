128             handler.config(config, true);
129         }
130 
131         handler.loadDatabase(Database.class);
132         session = handler.createSession();

182     /** Now, how to select Orders from a database by Criterions? <br/>
183      * The generated SQL code from this example
184      * will be similar like the next statement:
185      * <pre>
186      * SELECT * FROM item
187      * JOIN order ON order.id = item.id_order
188      * WHERE item.id >= 1
189      *   AND item.note LIKE '%table%'
190      *   AND order.note = 'My order';
191      * </pre>
192      * where both parameters are passed by a 'question mark' notation
193      * for a better security.
194      */
195     public void useSelect() {
196         Criterion<Item> crn = Item.ID.where(GE, 1L )
197             .and( Item.NOTE.where(CONTAINS, "table" ) )
198             .and( Item.ORDER.add(Order.NOTE).whereEq( "My order" ) )
199             ;
200 
201         for (Item item : session.createQuery(crn)) {
202             Date created = item.getOrder().getCreated(); // Lazy loading
203             System.out.println("Item: " + item + " // created: " + created);
204         }
205     }

207     /** Lern how to use the Criterion as an simple object validator only. */
208     public void useCriterions() {
209 
210         final Order order = new Order();
211         order.setId(100L);
212         order.setNote("my order");
213         order.setCreated(new Date());
214 
215         Criterion<Order> crnId, crnNote, crnCreated, crn;
216         crnId = Order.ID.where(GT, 99L);
217         crnNote = Order.NOTE.whereEq("another");
218         crnCreated = Order.CREATED.where(LE, new Date());
219         crn = null;
220 
221         // Simple condition: Order.ID>99
222         assert crnId.evaluate(order);
223 
224         // Compound condition: Order.ID>99 or Order.NOTE='another'
225         crn = crnId.or(crnNote);
226         assert crn.evaluate(order);
227 
228         // Compound condition with parentheses: Order.CREATED<=now() and (Order.NOTE='another' or Order.ID>99)
229         crn = crnCreated.and(crnNote.or(crnId));
230         assert crn.evaluate(order);
231 
232         // Another condition: (Order.CREATED<=now() or Order.NOTE='another') and Order.ID>99
233         crn = (crnCreated.or(crnNote)).and(crnId);
234         // ... or simple by a native priority:
235         crn =  crnCreated.or(crnNote).and(crnId);
236         assert crn.evaluate(order);
237     }

293     /** Use a 'native query' where the query is CREATED
294      * by a special entity signed by the @View annotation. <br/>
295      * Note the special <strong>inner parameter</strong> in the SQL statement on the Annotation of the class ViewOrder,
296      * where value for this (optional) parameter is set by the method Query.setSqlParameters();
297      * @see Query#setSqlParameters(java.lang.Object[])
298      */
299     public void useSelectWithNativeSQL() {
300         final Long excludedId = -7L;
301         SqlParameters sql = new SqlParameters().setSqlStatement
302                 ( "SELECT * FROM ("
303                 + "SELECT ord_order_alias.id"
304                 +         ", 1000 + count(*) AS item_count"
305                 + " FROM ${SCHEMA}.ord_order ord_order_alias"
306                 + " LEFT JOIN ${SCHEMA}.ord_item ord_item_alias"
307                 + " ON ord_order_alias.id = ord_item_alias.fk_order"
308                 + " WHERE ord_item_alias.id != ?" // Parameter is replaced by the excludedId
309                 + " GROUP BY ord_order_alias.id"
310                 + " ORDER BY ord_order_alias.id"
311                 + ") testView WHERE true"
312                 ).setParameters(excludedId);
313         Criterion<ViewOrder> crit = ViewOrder.ITEM_COUNT.whereLe(100);
314         long orderCount = session.createQuery(crit)
315                 .setSqlParameters(sql)
316                 .getCount()
317                 ;
318         System.out.println("Order Count: " + orderCount);
319 
320         Query<ViewOrder> orders = session.createQuery(crit)
321                 .setLimit(5)
322                 .orderBy(ViewOrder.ID)
323                 .setSqlParameters(sql)
324                 ;
325         for (ViewOrder order : orders) {
326             System.out.println("ORDER ROW: " + order);
327         }
328     }

502     /** Call a database stored procedure:
503      * <code>
504      * CREATE OR REPLACE FUNCTION db1.ujorm_test(integer, boolean) RETURNS integer
505      *    AS 'select $1 + $1;'
506      *    LANGUAGE SQL
507      *    IMMUTABLE
508      *    RETURNS NULL ON NULL INPUT;
509      * </code>
510      * Note: the source code is an aarly implementation prototype.<br>
511      * Note: test was running on the PostgreSQL release 8.4
512      */
513     public void useStoredProcedure() {
514         boolean isProcedureCreated = false; // create the database procedure first
515         if ( ! isProcedureCreated ) return;
516 
517         MyProcedure procedure = new MyProcedure();
518 
519         // Assign input parameters:
520         procedure.set(MyProcedure.PARAM_CODE, 5);
521         procedure.set(MyProcedure.PARAM_ENABLED, true);
522         Integer result = procedure.call(session); // Take the RESULT from the first parameter
523         System.out.println("The stored procedure result #1: " + result);
524 
525         // Another way how to get the output parameter:
526         procedure.set(MyProcedure.PARAM_CODE, 24);
527         procedure.set(MyProcedure.RESULT, null); // The output parameter(s) can't be initialized.
528         procedure.call(session);
529         result = procedure.get(MyProcedure.RESULT); // Take the RESULT from any output parameter
530         System.out.println("The stored procedure result #2: " + result);
531     }
